-- üåà Hi·ªáu ·ª©ng hover
local function hover(btn)
	btn.MouseEnter:Connect(function()
		TweenService:Create(btn, TweenInfo.new(0.2), {BackgroundColor3 = Color3.fromRGB(100, 0, 180)}):Play()
	end)
	btn.MouseLeave:Connect(function()
		TweenService:Create(btn, TweenInfo.new(0.2), {BackgroundColor3 = Color3.fromRGB(70, 0, 140)}):Play()
	end)
end

-- ‚úÖ T·∫£i script an to√†n
local function safeLoad(link)
	local ok, content = pcall(function()
		return game:HttpGet(link, true)
	end)
	if not ok then
		game.StarterGui:SetCore("SendNotification", {
			Title = "‚ùå BE HUB",
			Text = "Kh√¥ng t·∫£i ƒë∆∞·ª£c script: " .. tostring(link),
			Duration = 3
		})
		return
	end
	local success, err = pcall(function()
		loadstring(content)()
	end)
	if not success then
		game.StarterGui:SetCore("SendNotification", {
			Title = "‚ö†Ô∏è BE HUB",
			Text = "L·ªói khi ch·∫°y script: " .. tostring(err),
			Duration = 3
		})
	end
end
